[alias]
    s = status --short
    ss = status
    sb = status --short --branch
    bv = branch --verbose --verbose -a
    cob = checkout -b
    cpab = cherry-pick --abort
    mab = merge --abort
    rbab = rebase --abort
    rbam = rebase --ammend
    cull-local-branches = "!git checkout master && git branch --merged | xargs echo git branch --delete"
    cull-remote-branches = !"git branch --remotes --merged origin/master \
     | sed 's# *origin/##' | grep -v '^master$' xargs -I% git push origin :% 2>&1 | grep --colour=never 'deleted'"
